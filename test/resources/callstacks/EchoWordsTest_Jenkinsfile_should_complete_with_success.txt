   echo-words.run()
      echo-words.pipeline(groovy.lang.Closure)
         echo-words.agent(groovy.lang.Closure)
         echo-words.stages(groovy.lang.Closure)
            echo-words.stage(init, groovy.lang.Closure)
               echo-words.steps(groovy.lang.Closure)
                  echo-words.sh({ echo foo ; echo bar; } > words.txt)
                  echo-words.stash({name=words, includes=words.txt})
            echo-words.stage(grep, groovy.lang.Closure)
               echo-words.steps(groovy.lang.Closure)
                  echo-words.unstash({name=words})
                  echo-words.sh(grep foo words.txt)
                  echo-words.sh(grep bar words.txt)
